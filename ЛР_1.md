	 #include <iostream> 
	using namespace std;
	int main()
	{
		double x_n, x_k, x, shag, y;
		int n;
		cout << "enter x_n =";
		cin >> x_n;
		cout << "enter x_k =";
		cin >> x_k;
		cout << "enter n";
		cin >> n;
		shag = (x_k - x_n)/(n-1);
		for (int i = 0; i < n; ++i)
		{ x = x_n + shag * i;
			y = x * x;
			cout << "x["<<i<<"]="<< x;
			cout << "y["<< i <<"]="<< y << endl;
		}
		system("pause");
		return(0);
	}
