         #include <iostream>
         #include <iomanip>
         using namespace std;
         int main()
         {
             const int N = 3;
             const int M = 3;
             const int K = 3;
             int A[N][M];
             int B[M][K];
             int C[N][K];

              for (int i = 0; i < N; i++) 
              {
                  for (int j = 0; j < M; j++) 
                  {
                      cout << "A(" << i + 1 << " ," << j + 1 << ")=";
                      cin >> A[i][j];
                  }
              }

              for (int i = 0; i < N; i++) 
              {
                  for (int j = 0; j < M; j++) 
                  {
                      cout << setw(5) << A[i][j];
                  }
                  cout << endl;
              }

              for (int i = 0; i < M; i++) 
              {
                  for (int j = 0; j <K; j++) 
                  {
                      cout << "B(" << i + 1 << " ," << j + 1 << ")=";
                      cin >> B[i][j];
                  }
              }

              for (int i = 0; i < M; i++) 
              {
                  for (int j = 0; j < K; j++) 
                  {
                      cout << setw(5) << B[i][j];
                  }
                  cout << endl;
              }

              for (int i = 0; i < N; i++) 
              {
                  for (int j = 0; j < K; j++)
                  {
                      C[i][j] = 0;
                      for (int l = 0; l < M; l++) 
                      {
                          C[i][j] += A[i][l] * B[l][j];
                      }

                  }
              }
              cout << "C = " << endl;
              for (int i = 0; i < N; i++) 
              {
                  for (int j = 0; j < K; j++)
                  {
                      cout << setw(5) << C[i][j];
                  }
                  cout << endl;
              }
              system("PAUSE");
              return 0;

      }
